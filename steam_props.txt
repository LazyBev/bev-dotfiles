THIS IS MY STEAM LAUNCH OPTIONS (these are for performance)


# For VRChat           # DXVK_HUD is optional
unset SDL_VIDEODRIVER; DXVK_HUD=1 PROTON_USE_WINED3D=0 %command% -DX11 -novid -nojoy -no-cef -nod3d9ex -nohltv +cl_forcepreload 1

# Using Direct X11 (DXVK_HUD is optional)         # Use -F fsr if you have AMD gpu
DXVK_HUD=1 PROTON_USE_WINED3D=0 gamescope -f -r 0 -F nis --adaptive-sync -e --expose-wayland --rt --force-grab-cursor --threads 6 %command% -DX11 -novid -nojoy -no-cef -nod3d9ex -nohltv +cl_forcepreload 1

# Using Vulkan    # Use -F fsr if you have AMD gpu                                                        # Install mangoapp(mangoapp is optional)
gamescope -f -r 0 -F nis --adaptive-sync -e --expose-wayland --rt --steam --force-grab-cursor --threads 6 --mangoapp %command% -vulkan -novid -nojoy -no-cef -nod3d9ex -nohltv +cl_forcepreload 1
